def dict(list1):
    dictionary = {list1[i]: int(list1[i + 1]) for i in range(0, len(list1), 2)}
    return dictionary


#for sorting the list

def sort1(inplist):
    for i in range(1, len(inplist)):
        j = i-1
        next_element = inplist[i]

# Compare the current element with next element
		
        while (inplist[j] > next_element) and (j >= 0):
            inplist[j+1] = inplist[j]
            j=j-1
        inplist[j+1] = next_element
        
#main function
